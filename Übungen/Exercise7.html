function identity_function(x) {
return function() {
return x;
}
}

function addf(x) {
return function(y) {
return x + y;
}
}

function applyf(binaryFn) {
return function(x) {
return function(y) {
return binaryFn(x, y);
}
}
}

function curry(binaryFn, x) {
return function(y) {
return binaryFn(x, y);
}
}

//addf
var inc = addf(1);
//applyf
var inc = applyf(function(x, y) { return x + y; })(1);
//curry
var inc = curry(function(x, y) { return x + y; }, 1);

function methodize(binaryFn) {
return function(y) {
return binaryFn(this, y);
}
}

function demethodize(unaryMethod) {
return function(x, y) {
return unaryMethod.call(x, y);
}
}

function twice(binaryFn) {
return function(x) {
return binaryFn(x, x);
}
}

function composeu(fn1, fn2) {
return function(x) {
return fn2(fn1(x));
}
}

function composeb(fn1, fn2) {
return function(x, y, z) {
return fn2(fn1(x, y), z);
}
}

function once(fn) {
var called = false;
return function() {
if (!called) {
called = true;
return fn.apply(this, arguments);
} else {
throw new Error("Function can only be called once.");
}
}
}

function counterf(start) {
var count = start;
return {
inc: function() {
count++;
return count;
},
dec: function() {
count--;
return count;
}
}
}

function revocable(fn) {
var revoker = false;
return {
invoke: function() {
if (!revoker) {
return fn.apply(this, arguments);
} else {
throw new Error("Function has been revoked.");
}
},
revoke: function() {
revoker = true;
}
}
}

function vector() {
var data = [];
return {
get: function(index) {
return data[index];
},
store: function(index, value) {
data[index] = value;
},
append: function(value) {
data.push(value);
}
}
}
