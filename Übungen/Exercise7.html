        function once(fn) {
            var called = false;
            return function() {
              if (!called) {
                called = true;
                return fn.apply(this, arguments);
              } else {
                throw new Error("Function can only be called once.");
              }
            }
          }
      
        function counterf(start) {
            var count = start;
            return {
              inc: function() {
                count++;
                return count;
              },
              dec: function() {
                count--;
                return count;
              }
            }
          }
      
        function revocable(fn) {
            var revoker = false;
            return {
              invoke: function() {
                if (!revoker) {
                  return fn.apply(this, arguments);
                } else {
                  throw new Error("Function has been revoked.");
                }
              },
              revoke: function() {
                revoker = true;
              }
            }
          }
